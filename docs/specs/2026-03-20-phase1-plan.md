# Eventuous Go — Phase 1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Implement the core Eventuous framework in idiomatic Go with KurrentDB integration — functional and aggregate command services, subscriptions, and an adequate mirror of the .NET test suite.

**Architecture:** Multi-module Go project. `core/` has near-zero deps and contains domain, persistence, command, and subscription abstractions. `kurrentdb/` implements the store and subscriptions against EventStoreDB/KurrentDB. `otel/` adds tracing/metrics. Functional command service is primary; aggregate is optional.

**Tech Stack:** Go 1.23+, `github.com/google/uuid`, `github.com/EventStore/EventStore-Client-Go/v4` (KurrentDB module), `go.opentelemetry.io/otel` (OTel module), `github.com/testcontainers/testcontainers-go` (integration tests).

**Spec:** `docs/specs/2026-03-20-phase1-design.md`

**Test strategy:** Mirror the .NET Eventuous test suite structure. The .NET codebase uses conformance base test classes (`StoreAppendTests<T>`, `StoreReadTests<T>`, `SubscribeToStreamBase<T>`, etc.) that each store implementation inherits. The Go port must replicate this pattern: exported test functions in `core/test/storetest/` and `core/test/subtest/` that any implementation calls with its concrete store. Additionally, a shared test domain (`core/test/testdomain/`) provides Booking events, state, and fold — mirroring `Eventuous.Sut.Domain` and `Eventuous.Sut.App` from .NET. This is an exported package (not `internal/`) so that the `kurrentdb/` module can import it for integration tests.

```go
package eventuous

import (
    "errors"
    "strings"
)

// StreamName is a typed string with "{Category}-{ID}" convention.
type StreamName string

// NewStreamName creates a stream name from category and entity ID.
func NewStreamName(category, id string) StreamName {
    return StreamName(category + "-" + id)
}

// Category returns everything before the first "-".
func (s StreamName) Category() string {
    if i := strings.IndexByte(string(s), '-'); i >= 0 {
        return string(s[:i])
    }
    return string(s)
}

// ID returns everything after the first "-".
func (s StreamName) ID() string {
    if i := strings.IndexByte(string(s), '-'); i >= 0 {
        return string(s[i+1:])
    }
    return ""
}

func (s StreamName) String() string { return string(s) }

// ExpectedVersion for optimistic concurrency on append.
type ExpectedVersion int64

const (
    VersionNoStream ExpectedVersion = -1 // stream must not exist
    VersionAny      ExpectedVersion = -2 // no concurrency check
)

// ExpectedState controls how the command service loads state.
type ExpectedState int

const (
    IsNew      ExpectedState = iota // stream must not exist
    IsExisting                       // stream must exist
    IsAny                            // either
)

// Metadata is event metadata.
type Metadata map[string]any

const (
    MetaCorrelationID = "eventuous.correlation-id"
    MetaCausationID   = "eventuous.causation-id"
    MetaMessageID     = "eventuous.message-id"
)

func (m Metadata) CorrelationID() string {
    if v, ok := m[MetaCorrelationID].(string); ok {
        return v
    }
    return ""
}

func (m Metadata) WithCorrelationID(id string) Metadata {
    c := m.clone()
    c[MetaCorrelationID] = id
    return c
}

func (m Metadata) CausationID() string {
    if v, ok := m[MetaCausationID].(string); ok {
        return v
    }
    return ""
}

func (m Metadata) WithCausationID(id string) Metadata {
    c := m.clone()
    c[MetaCausationID] = id
    return c
}

func (m Metadata) clone() Metadata {
    c := make(Metadata, len(m)+1)
    for k, v := range m {
        c[k] = v
    }
    return c
}

// Sentinel errors.
var (
    ErrStreamNotFound        = errors.New("eventuous: stream not found")
    ErrOptimisticConcurrency = errors.New("eventuous: wrong expected version")
    ErrAggregateNotFound     = errors.New("eventuous: aggregate not found")
    ErrHandlerNotFound       = errors.New("eventuous: command handler not found")
)
```

`core/codec/typemap.go`:
```go
package codec

import (
    "fmt"
    "reflect"
    "sync"
)

// TypeMap holds bidirectional type ↔ name mappings.
type TypeMap struct {
    mu       sync.RWMutex
    toName   map[reflect.Type]string
    fromName map[string]reflect.Type
}

// NewTypeMap creates an empty type map.
func NewTypeMap() *TypeMap {
    return &TypeMap{
        toName:   make(map[reflect.Type]string),
        fromName: make(map[string]reflect.Type),
    }
}

// Register maps a Go type to a persistent event type name.
func Register[E any](tm *TypeMap, name string) error {
    var zero E
    t := reflect.TypeOf(zero)
    tm.mu.Lock()
    defer tm.mu.Unlock()

    if existing, ok := tm.toName[t]; ok {
        if existing != name {
            return fmt.Errorf("codec: type %v already registered as %q, cannot register as %q", t, existing, name)
        }
        return nil
    }
    if existing, ok := tm.fromName[name]; ok {
        if existing != t {
            return fmt.Errorf("codec: name %q already registered for %v, cannot register for %v", name, existing, t)
        }
        return nil
    }

    tm.toName[t] = name
    tm.fromName[name] = t
    return nil
}

// TypeName returns the registered name for an event value.
func (tm *TypeMap) TypeName(event any) (string, error) {
    t := reflect.TypeOf(event)
    tm.mu.RLock()
    defer tm.mu.RUnlock()
    if name, ok := tm.toName[t]; ok {
        return name, nil
    }
    return "", fmt.Errorf("codec: unregistered event type %v", t)
}

// NewInstance creates an empty pointer of the type registered under name.
func (tm *TypeMap) NewInstance(name string) (any, error) {
    tm.mu.RLock()
    defer tm.mu.RUnlock()
    t, ok := tm.fromName[name]
    if !ok {
        return nil, fmt.Errorf("codec: unknown event type name %q", name)
    }
    return reflect.New(t).Interface(), nil
}

```

Note: `ReflectTypeMap()` (auto-registers types by struct name on first use) is **deferred** — it requires a custom wrapper. Explicit registration via `Register[E]()` is the only path for Phase 1.

```go
package codec

import (
    "encoding/json"
    "reflect"
)

// JSONCodec serializes events as JSON using a TypeMap for type resolution.
type JSONCodec struct {
    types *TypeMap
}

// NewJSON creates a JSON codec backed by the given type map.
func NewJSON(types *TypeMap) *JSONCodec {
    return &JSONCodec{types: types}
}

func (c *JSONCodec) Encode(event any) ([]byte, string, string, error) {
    name, err := c.types.TypeName(event)
    if err != nil {
        return nil, "", "", err
    }
    data, err := json.Marshal(event)
    if err != nil {
        return nil, "", "", err
    }
    return data, name, "application/json", nil
}

func (c *JSONCodec) Decode(data []byte, eventType string) (any, error) {
    ptr, err := c.types.NewInstance(eventType)
    if err != nil {
        return nil, err
    }
    if err := json.Unmarshal(data, ptr); err != nil {
        return nil, err
    }
    // Dereference pointer to return value type
    return reflect.ValueOf(ptr).Elem().Interface(), nil
}
```
